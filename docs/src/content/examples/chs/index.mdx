```ts
scene.draw((p) => {
  const { radius: r, thickness: t } = p;
  const ir = r - t;

  const path = geom.path();

  // Outer circle — 4 quarter arcs
  path
    .moveToAbs(-r, 0)
    .arcTo(r, r, r, true)
    .arcTo(r, -r, r, true)
    .arcTo(-r, -r, r, true)
    .arcTo(-r, r, r, true);

  // Inner circle — same winding
  path
    .moveToAbs(-ir, 0)
    .arcTo(ir, ir, ir, true)
    .arcTo(ir, -ir, ir, true)
    .arcTo(-ir, -ir, ir, true)
    .arcTo(-ir, ir, ir, true);

  // Center mark + radius callout
  const dim = dims.dimension();
  const angle = Math.PI / 4;
  dim.moveToAbs(0, 0).centerMark(20);
  dim
    .moveToAbs(0, 0)
    .lineToAbs(r * Math.cos(angle), r * Math.sin(angle))
    .arrowFilled(angle)
    .textAtAbs(r / 2 - 10, r / 2, `R${r}mm`, "end");
});
```
