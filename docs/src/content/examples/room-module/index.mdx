```typescript
import { Pluton2D } from "pluton-2d";

type SceneParams = {
  roomWidth: number;
  roomDepth: number;
  wallThickness: number;
  doorWidth: number;
  windowWidth: number;
  windowOffset: number;
};

const svgElement = document.querySelector("svg#room-module") as SVGSVGElement;
const scene = new Pluton2D<SceneParams>(svgElement, {
  roomWidth: 560,
  roomDepth: 430,
  wallThickness: 26,
  doorWidth: 94,
  windowWidth: 150,
  windowOffset: 0,
});

const geometry = scene.geometry.group();
const dimensions = scene.dimensions.group();
const furnitureFillId = scene.addHatchFill("#475569", 0.12);

scene.draw((params) => {
  const roomWidthValue = params.roomWidth;
  const roomDepthValue = params.roomDepth;
  const wallThicknessValue = Math.max(16, Math.min(params.wallThickness, 46));

  const outerLeftX = -roomWidthValue / 2;
  const outerRightX = roomWidthValue / 2;
  const outerBottomY = -roomDepthValue / 2;
  const outerTopY = roomDepthValue / 2;
  const innerLeftX = outerLeftX + wallThicknessValue;
  const innerBottomY = outerBottomY + wallThicknessValue;

  const wallsPath = geometry.path({ className: "demo-room-walls" });
  wallsPath
    .moveToAbs(outerLeftX, outerBottomY)
    .lineToAbs(outerRightX, outerBottomY)
    .lineToAbs(outerRightX, outerTopY)
    .lineToAbs(outerLeftX, outerTopY)
    .close();

  const windowPath = geometry.path({ className: "demo-room-window" });
  windowPath.moveToAbs(-params.windowWidth / 2, outerTopY - wallThicknessValue * 0.34).lineToAbs(params.windowWidth / 2, outerTopY - wallThicknessValue * 0.34);

  const doorPath = geometry.path({ className: "demo-room-door" });
  const doorWidthValue = params.doorWidth;
  const doorLeftX = innerLeftX + 48;
  doorPath
    .moveToAbs(doorLeftX, innerBottomY)
    .lineToAbs(doorLeftX + doorWidthValue, innerBottomY)
    .moveToAbs(doorLeftX, innerBottomY)
    .lineToAbs(doorLeftX, innerBottomY + doorWidthValue);

  const furniturePath = geometry.path({ className: "demo-room-furniture", fill: furnitureFillId });
  furniturePath.moveToAbs(-90, -20).lineToAbs(90, -20).lineToAbs(90, 70).lineToAbs(-90, 70).close();

  const dimensionPath = dimensions.dimension({ className: "room-dim" });
  dimensionPath.moveToAbs(outerLeftX, outerTopY + 30).lineTo(roomWidthValue, 0);
  dimensionPath.moveToAbs(outerLeftX - 34, outerBottomY).lineTo(0, roomDepthValue);
});
```
